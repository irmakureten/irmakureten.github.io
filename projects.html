Problem 1: Basic Math Operations

Write a program that asks the user to enter two numbers.
The program should calculate and display:

  • The sum
  • The difference
  • The product
  • The quotient (if the second number is not zero; otherwise, show a warning)

Sample Output:

Enter the first number: 12
Enter the second number: 5
First number: 12.0
Second number: 5.0
Sum: 17.0
Difference: 7.0
Product: 60.0
Quotient: 2.4
        
import java.util.Scanner;
public class MyProgram {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);

    int c1 = scanner.nextInt(), v1 = scanner.nextInt(), s1 = scanner.nextInt();
    int c2 = scanner.nextInt(), v2 = scanner.nextInt(), s2 = scanner.nextInt();
    int c3 = scanner.nextInt(), v3 = scanner.nextInt(), s3 = scanner.nextInt();

    double total = c1 + v1 + s1 + c2 + v2 + s2 + c3 + v3 + s3;
    int totalInt = (int) total;
    double avg = total / 3;
    int avgInt = totalInt / 3;

    System.out.println("Total sales: " + total);
    System.out.println("Average sales (decimal): " + avg);
    System.out.println("Average sales (integer): " + avgInt);

    if (avg >= 100) System.out.println("Performance Grade: A");
    else if (avg >= 70) System.out.println("Performance Grade: B");
    else if (avg >= 50) System.out.println("Performance Grade: C");
  }
}

Problem 2: Ice Cream Sales Tracker

You're running an ice cream shop. Get 3-day sales for 3 flavors and calculate:

  • Total sales
  • Average sales (decimal + integer)
  • Performance grade (A/B/C) — no if-else chains

Sample Output:

Day 1 - Chocolate: 30
Day 2 - Vanilla: 30
Day 3 - Strawberry: 30
Total sales: 300.0
Average sales (decimal): 100.0
Average sales (integer): 100
Performance Grade: A
        
import java.util.Scanner;
public class MyProgram {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);

    int mass = scanner.nextInt();
    int force = scanner.nextInt();
    int burn = scanner.nextInt();

    double a = (double) force / mass;
    double v = a * burn;
    double alt = 0.5 * a * burn * burn;

    System.out.println("Net acceleration: " + a);
    System.out.println("Final velocity: " + v);
    System.out.println("Altitude reached: " + alt);

    System.out.println("  /\\  ");
    System.out.println(" /  \\ ");
    System.out.println(" |==| ");
    System.out.println("/____\\");
  }
}

Problem 3: Rocket to the Stars

The user enters rocket mass (kg), engine force (N), and burn time (s). Calculate:

  • Acceleration
  • Final velocity
  • Altitude

Sample Output:

Enter rocket mass (kg): 1000
Enter engine force (N): 20000
Enter burn time (s): 10

Net acceleration (m/s²): 20.0
Final velocity: 200.0
Altitude: 1000.0
        
import java.util.Scanner;
public class MyProgram {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);

    double N = scanner.nextDouble();
    double x0 = N / 2.0;
    double x1 = 0.5 * (x0 + N / x0);
    double x2 = 0.5 * (x1 + N / x1);
    double x3 = 0.5 * (x2 + N / x2);

    System.out.println("x0 = " + x0);
    System.out.println("x1 = " + x1);
    System.out.println("x2 = " + x2);
    System.out.println("x3 = " + x3);
    System.out.println("Approximate sqrt: " + x3);
  }
}

import java.util.Scanner;
public class MyProgram {
  public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);

    int m = scanner.nextInt();
    int d = scanner.nextInt();
    int y = scanner.nextInt();

    int y0 = y - (14 - m) / 12;
    int x = y0 + y0 / 4 - y0 / 100 + y0 / 400;
    int m0 = m + 12 * ((14 - m) / 12) - 2;
    int d0 = (d + x + (31 * m0) / 12) % 7;

    System.out.println("Day of week (0=Sun,...6=Sat): " + d0);
  }
}
